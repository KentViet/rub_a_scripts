def construct_arguments(*args)

args.each_with_index do |item, index|
  if item.class == Array
    item.each_with_index do |inner_item, index|
      if inner_item.class == Array
        puts "this is an array: #{inner_item.class}"
        construct_arguments(inner_item)
      else
        puts "this is not an array: #{inner_item}"
      end
    end
  else
    puts item
  end
end

end

construct_arguments(@my_array3) construct_arguments(@not_array) construct_arguments(@my_str)
